use StudentData
create procedure UpdateStudent1
(
  @strollno int,
  @stname varchar(50),
  @stfname varchar(50),
  @stcity varchar(50)
)
as
begin 
update studentinfo set sname=@stname, fname=@stfname, city=@stcity where srollno=@strollno
end
