--------------------
        [HttpPost]

        public ActionResult Index(string gender)
        {
            ViewBag.result ="You Have Selected "+gender.ToString();
            if(gender=="Male")
            {
                ViewBag.maleoption = "checked";
                ViewBag.femaleoption = "";
            }
            if (gender == "Female")
            {
                ViewBag.maleoption = "";
                ViewBag.femaleoption = "checked";
            }

            return View();
        }
