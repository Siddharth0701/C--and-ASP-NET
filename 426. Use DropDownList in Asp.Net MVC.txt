Controller Coding ( HomeController )
----------------------------------

 public ActionResult Index()
        {
            ViewBag.dropdowndata = "";
            return View();
        }
        [HttpPost]
        public ActionResult Index(FormCollection form)
        {
        
            ViewBag.dropdowndata = "You Have Selected Profession "+form["profession"].ToString();
            return View();
        }

----------------------------------
